// 1. 错误边界组件
class ErrorBoundary extends React.Component<
  { children: React.ReactNode; fallback?: React.ReactNode },
  { hasError: boolean; error?: Error }
> {
  constructor(props: { children: React.ReactNode; fallback?: React.ReactNode }) {
    super(props)
    this.state = { hasError: false }
  }

  static getDerivedStateFromError(error: Error) {
    return { hasError: true, error }
  }

  componentDidCatch(error: Error, errorInfo: React.ErrorInfo) {
    console.error('ErrorBoundary caught an error:', error, errorInfo)
    reportError(error, errorInfo)
  }

  render() {
    if (this.state.hasError) {
      return this.props.fallback || <DefaultErrorFallback error={this.state.error} />
    }
    return this.props.children
  }
}

// 2. 统一的API错误处理
class ApiError extends Error {
  constructor(
    public code: number,
    message: string,
    public details?: any
  ) {
    super(message)
    this.name = 'ApiError'
  }
}

// 3. 错误提示钩子
function useErrorHandler() {
  const toast = useToast()
  
  const handleError = useCallback((error: unknown) => {
    if (error instanceof ApiError) {
      toast({
        title: getErrorMessage(error.code),
        description: error.message,
        variant: 'destructive',
      })
      
      if (error.code === 40100) {
        // 处理未授权
        redirectToLogin()
      }
    } else if (error instanceof Error) {
      toast({
        title: '发生错误',
        description: error.message,
        variant: 'destructive',
      })
    }
  }, [toast])
  
  return { handleError }

import (
    "fmt"

type AppError struct {

    Message    string      `json:"message"`

    Err        error       `json:"-"`
    StackTrace string      `json:"-"`

    if e.Err != nil {
        return fmt.Sprintf("%s: %v", e.Message, e.Err)

func ErrorHandler() gin.HandlerFunc {

import "fmt"

// 缓存Key常量
const (
    KeyProject      = "project:%d"
    KeyProjectList  = "projects:page:%d:size:%d"
    KeyProjectTasks = "project:%d:tasks"
    KeyUser         = "user:%d"
)

// 生成缓存Key

// 2. 缓存更新策略
type CacheManager struct {
    redis *redis.Client
    local *bigcache.BigCache
}

// Cache-Aside模式读取
func (m *CacheManager) GetOrSet(ctx context.Context, key string, ttl time.Duration, fn func() (interface{}, error)) (interface{}, error) {
    // 先查本地缓存
    if data, err := m.local.Get(key); err == nil {
        return data, nil
    }
    
    // 再查Redis
    data, err := m.redis.Get(ctx, key).Result()
    if err == nil {
        // 回填本地缓存
        m.local.Set(key, []byte(data))
        return data, nil
    }
    
    // 缓存未命中，调用函数获取
    result, err := fn()
    if err != nil {
        return nil, err
    }
    
    // 序列化并缓存
    jsonData, _ := json.Marshal(result)
    m.redis.Set(ctx, key, jsonData, ttl)
    m.local.Set(key, jsonData)
    
    return result, nil
}

// 3. 缓存失效策略
func (m *CacheManager) InvalidateProject(ctx context.Context, id int64) {
    // 清除项目详情缓存
    m.redis.Del(ctx, ProjectKey(id))
    
    // 清除相关列表缓存（简化处理，实际可以使用标签化缓存）
    pattern := "projects:page:*"
    keys, _ := m.redis.Keys(ctx, pattern).Result()
    if len(keys) > 0 {
        m.redis.Del(ctx, keys...)
    }
    
    // 清除本地缓存（全量清除，简化处理）
    m.local.Reset()
}

// 1. 限流中间件
package middleware

import (
    "github.com/gin-gonic/gin"
    "github.com/redis/go-redis/v9"
    "golang.org/x/time/rate"
    "net/http"
    "sync"
    "time"
)

type RateLimiter struct {
    redis   *redis.Client
    limiters map[string]*rate.Limiter
    mu      sync.RWMutex
}

func NewRateLimiter(redis *redis.Client) *RateLimiter {
    return &RateLimiter{
        redis:    redis,
        limiters: make(map[string]*rate.Limiter),
    }
}

// 2. 基于IP的限流
func (rl *RateLimiter) IPRateLimit(rps int) gin.HandlerFunc {
    return func(c *gin.Context) {
        ip := c.ClientIP()
        
        rl.mu.Lock()
        limiter, exists := rl.limiters[ip]
        if !exists {
            limiter = rate.NewLimiter(rate.Limit(rps), rps*2)
            rl.limiters[ip] = limiter
        }
        rl.mu.Unlock()
        
        if !limiter.Allow() {
            c.JSON(http.StatusTooManyRequests, gin.H{
                "code":    42900,
                "message": "请求过于频繁，请稍后再试",
            })
            c.Abort()
            return
        }
        
        c.Next()
    }
}

// 3. 基于用户的限流
func (rl *RateLimiter) UserRateLimit(rps int) gin.HandlerFunc {
    return func(c *gin.Context) {
        userID := c.GetString("user_id")
        if userID == "" {
            c.Next()
            return
        }
        
        key := fmt.Sprintf("user:%s:rate_limit", userID)
        
        // 使用Redis滑动窗口
        ctx := c.Request.Context()
        now := time.Now().Unix()
        window := 60 // 60秒窗口
        limit := rps * window
        
        // 移除窗口外的记录
        rl.redis.ZRemRangeByScore(ctx, key, "0", fmt.Sprintf("%d", now-int64(window)))
        
        // 获取当前窗口内的请求数
        count, _ := rl.redis.ZCard(ctx, key).Result()
        
        if count >= int64(limit) {
            c.JSON(http.StatusTooManyRequests, gin.H{
                "code":    42900,
                "message": "请求过于频繁，请稍后再试",
            })
            c.Abort()
            return
        }
        
        // 添加当前请求
        rl.redis.ZAdd(ctx, key, redis.Z{Score: float64(now), Member: now})
        rl.redis.Expire(ctx, key, time.Duration(window)*time.Second)
        
        c.Next()
    }
}
